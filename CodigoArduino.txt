#include <Wire.h>
#include <Adafruit_RGBLCDShield.h>

Adafruit_RGBLCDShield lcd = Adafruit_RGBLCDShield();

void setup() {
  Serial.begin(115200);
  lcd.begin(16, 2);            // LCD de 16 columnas x 2 filas
  lcd.setBacklight(0x7);       // Encender retroiluminaciÃ³n
  lcd.setCursor(0, 0);
  lcd.print("Esperando...");
}

void loop() {
  if (Serial.available() >= 8) {
    byte buffer[8];
    Serial.readBytes(buffer, 8);

    int coordX = (buffer[0] << 24) | (buffer[1] << 16) | (buffer[2] << 8) | buffer[3];
    int coordY = (buffer[4] << 24) | (buffer[5] << 16) | (buffer[6] << 8) | buffer[7];

    // Mostrar en el LCD
    lcd.clear();
    lcd.setCursor(0, 0);
    lcd.print("X: ");
    lcd.print(coordX);
    lcd.setCursor(0, 1);
    lcd.print("Y: ");
    lcd.print(coordY);

    // Solo como debug por consola:
    Serial.print("Recibido X: ");
    Serial.print(coordX);
    Serial.print("  Y: ");
    Serial.println(coordY);
  }
}
